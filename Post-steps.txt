#Initialize cluster with kubeadm
kubeadm init --pod-network-cidr=192.168.0.0/16 --kubernetes-version=1.32.0

mkdir -p $HOME/.kube
cp -i /etc/kubernetes/admin.conf $HOME/.kube/config
chown ansible:ansible /home/ansible/.kube/config

#Untaint the control plane
kubectl taint nodes --all node-role.kubernetes.io/control-plane-

#Install Calico
kubectl create -f https://raw.githubusercontent.com/projectcalico/calico/v3.29.1/manifests/tigera-operator.yaml
kubectl create -f https://raw.githubusercontent.com/projectcalico/calico/v3.29.1/manifests/custom-resources.yaml

cat <<EOF | sudo tee /tmp/join.sh
#!/bin/bash
kubeadm token create --print-join-command
EOF
scp /tmp/join.sh 172.31.1.2:/tmp/


#Join worker node
sleep 30
chmod +x /tmp/join.sh
/tmp/join.sh | bash


Ansible master
1) Execute kubeadm init
2) Create a directory
3) Copy a file locally
4) Change permissions
5) Execute kubectl
6) Create a file with content
7) Copy a file to another machine

Ansible worker
1) Execute a command
2) Change permissions
3) Execute a script
